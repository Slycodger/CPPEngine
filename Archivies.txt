
//Creates text file, allows you to write to it and then reads what you wrote.
#include <iostream>
#include <fstream>
#include <string>
int main()
{
	std::string Words;
	char Output[1024];
	std::getline(std::cin >> std::ws, Words);
	std::fstream File{ "Test.txt", std::ios::binary | std::ios::trunc | std::ios::in | std::ios::out };
	File.write(Words.c_str(), Words.length());
	File.seekg(0, File.end);
	int Size = File.tellg();
	File.seekg(0, File.beg);
	if (!File.is_open())
		std::cout << "Couldn't open\n";
	File.read(Output, Size);
	for (int i = 0; i < Size; i++)
		std::cout << Output[i];
}
